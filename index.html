# 快速开始

# 安装组件库

npm i supe-ui

# 引用组件库

在 main.js 中引用组件库

// 全部引入
import 'supe-ui/dist/css/index.css';
import SUI from 'supe-ui';
Vue.use(SUI);


// 按需引入
import 'supe-ui/dist/css/demo.css';
import { Demo } from 'supe-ui';
Vue.use(Demo);
